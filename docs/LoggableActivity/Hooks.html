def log(action, actor: nil, params: {})
  @action = action
  @actor = actor || Thread.current[:current_user]
  # LoggableActivity::EncryptionKey.for_record(self)

  return if @actor.nil?

  @record = self
  @params = params
  @payloads = []

  case action
  when :create, :show
    log_activity
  when :destroy
    log_destroy
  when :update
    log_update
  else
    log_custom_activity(action)
  end
end
